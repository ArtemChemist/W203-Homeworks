\item Correlation between $X$ and $aX+b$

\begin{align*}
Corr[aX+b,X] = \frac{Cov[aX+b,X]}{\sqrt{Var[aX+b]}*\sqrt{Var[X]}}\\[1em] \frac{a*Cov[X,X]}{{|a|}*\sqrt{Var[X]}*\sqrt{Var[X]}} = \\[1em]
\frac{a}{|a|}*Corr[X,X] = \frac{a}{|a|} =\\[1em]
\begin{cases}
1, a>0\\
-1, a<0\\
\end{cases}  
\end{align*}
